workspace {

    model {
        alumno = person "Alumno" "Consume contenido, entrega tareas y revisa recomendaciones"
        docente = person "Docente" "Publica contenido, crea tareas y califica entregas"
        admin = person "Administrador" "Gestiona catalogo y operacion academica"

        lms = softwareSystem "Plataforma LMS" "Sistema web para el curso de Estructuras de Datos" {
            web = container "Frontend Web" "Interfaz para alumnos y docentes" "Angular"
            api = container "API LMS" "Logica de negocio y endpoints REST" "Spring Boot"
            rec = container "Servicio de Recomendaciones" "Sugiere siguiente clase o refuerzo" "Spring Service"
            batch = container "Carga Inicial" "Carga curso, clases y recursos" "Spring Batch"
            db = container "Base de Datos" "Datos academicos y progreso" "PostgreSQL"
            storage = container "Almacenamiento" "Archivos de contenido y entregas" "File Storage"
        }

        alumno -> web "Navega el curso y entrega tareas"
        docente -> web "Administra contenido y calificaciones"
        admin -> web "Supervisa configuracion academica"

        web -> api "REST/JSON"
        api -> db "CRUD de cursos, clases, tareas, entregas y notas"
        api -> storage "Lectura/escritura de archivos"
        api -> rec "Solicita recomendacion por alumno"
        rec -> db "Consulta avance, notas y puntualidad"
        batch -> db "Carga inicial de datos"
        batch -> storage "Carga recursos iniciales"
    }

    views {
        systemContext lms "contexto" {
            include alumno
            include docente
            include admin
            include lms
            autoLayout lr
            title "Contexto del sistema LMS"
        }

        container lms "contenedores" {
            include *
            autoLayout lr
            title "Vista de contenedores LMS"
        }

        dynamic lms "escenario_alumno" {
            title "Escenario principal: consumir contenido, entregar tarea y recibir recomendacion"
            alumno -> web "1. Entra al curso"
            web -> api "2. Solicita clases y contenido"
            api -> db "3. Consulta progreso e historial"
            api -> storage "4. Obtiene recursos (PDF, video, imagen, enlaces)"
            alumno -> web "5. Sube entrega"
            web -> api "6. Registra entrega"
            api -> db "7. Guarda entrega y nota"
            api -> rec "8. Pide recomendacion"
            rec -> db "9. Evalua avance y rendimiento"
            rec -> api "10. Retorna sugerencia"
            api -> web "11. Muestra nota y recomendacion"
            autoLayout lr
        }

        theme default
    }
}
