2. Gestión de memoria

// Destructor que libera toda la memoria
~BST() { liberarArbol(raiz); }

void liberarArbol(Nodo* r) {
    if (!r) return;
    liberarArbol(r->izq);
    liberarArbol(r->der);
    delete r;     // postorden: liberar hojas primero
}

3. Pruebas sistemáticas

void probarBST() {
    BST arbol;
    // Prueba 1: árbol vacío
    assert(!arbol.buscar(5));
    assert(arbol.getAltura() == 0);

    // Prueba 2: inserción y búsqueda
    arbol.insertar(10); arbol.insertar(5); arbol.insertar(15);
    assert(arbol.buscar(5) == true);
    assert(arbol.buscar(99) == false);

    // Prueba 3: eliminación
    arbol.eliminar(5);
    assert(!arbol.buscar(5));

    // Prueba 4: inserción ordenada (peor caso BST)
    BST arbolDeg;
    for (int i=1; i<=10; i++) arbolDeg.insertar(i);
    assert(arbolDeg.getAltura() == 10);  // degenerado

    cout << "Todas las pruebas pasaron.\n";
}
