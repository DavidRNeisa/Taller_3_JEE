Clase 11: Recursión

Dividir para conquistar: funciones que se resuelven a sí mismas

Estructura de toda función recursiva

  • Caso base: condición que detiene la recursión y devuelve un valor directo.
  • Caso recursivo: reduce el problema y llama a la función con entrada más pequeña.
Sin caso base → recursión infinita → stack overflow.

Factorial

// 0!=1;  n!=n×(n-1)!
int factorial(int n) {
    if (n==0) return 1;           // caso base
    return n * factorial(n-1);    // caso recursivo
}
// factorial(4) = 4 × 3 × 2 × 1 × 1 = 24
// Complejidad: O(n) tiempo, O(n) espacio (pila de llamadas)

Fibonacci: el peligro de la recursión ingenua

int fib(int n) {             // O(2ⁿ) — inaceptable para n grande
    if (n <= 1) return n;
    return fib(n-1) + fib(n-2);
}

// Solución: memoización → O(n)
unordered_map<int,int> memo;
int fibM(int n) {
    if (n <= 1) return n;
    if (memo.count(n)) return memo[n];
    return memo[n] = fibM(n-1) + fibM(n-2);
}

Torres de Hanoi

void hanoi(int n, char orig, char dest, char aux) {
    if (n==1) { cout<<"Mover disco 1: "<<orig<<"→"<<dest<<"\n"; return; }
    hanoi(n-1, orig, aux, dest);
    cout << "Mover disco " << n << ": " << orig << "→" << dest << "\n";
    hanoi(n-1, aux, dest, orig);
}
// hanoi(3,'A','C','B') → 2³-1 = 7 movimientos
// Complejidad: O(2ⁿ)

Búsqueda binaria recursiva

int binaria(int arr[], int izq, int der, int x) {
    if (izq>der) return -1;          // caso base: no encontrado
    int m = (izq+der)/2;
    if (arr[m]==x) return m;         // caso base: encontrado
    if (x < arr[m]) return binaria(arr, izq, m-1, x);
    else             return binaria(arr, m+1, der, x);
}
// O(log n) tiempo, O(log n) espacio

Recorrer un árbol (anticipando el tema)

// La recursión es natural para estructuras recursivas como árboles
void inorden(Nodo* raiz) {
    if (!raiz) return;
    inorden(raiz->izq);
    cout << raiz->valor << " ";
    inorden(raiz->der);
}
