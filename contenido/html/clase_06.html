Clase 6: La STL de C++

Standard Template Library: estructuras y algoritmos listos para usar

¿Qué es la STL?

La Standard Template Library es el conjunto de clases y funciones genéricas incluidas en el estándar C++. Sus tres pilares: contenedores (almacenan datos), iteradores (recorren contenedores) y algoritmos (operan sobre rangos).

Contenedores secuenciales

#include <vector>   // arreglo dinámico
#include <list>     // lista doblemente enlazada
#include <deque>    // cola doble

vector<int> v = {3,1,4,1,5};
v.push_back(9);            // O(1) amortizado
v.insert(v.begin(), 0);    // O(n) — desplaza elementos
cout << v[2];              // O(1) — acceso directo

list<string> l;
l.push_front("Zoe");       // O(1) — sin mover elementos
l.push_back("Mario");      // O(1)

Adaptadores (Pila, Cola, Cola de Prioridad)

#include <stack>
#include <queue>

stack<int> pila;
pila.push(10); pila.push(20);
cout << pila.top();   // 20

queue<int> cola;
cola.push(10); cola.push(20);
cout << cola.front(); // 10

priority_queue<int> maxHeap;
maxHeap.push(5); maxHeap.push(1); maxHeap.push(8);
cout << maxHeap.top(); // 8 — el mayor siempre en la cima

Contenedores asociativos

#include <map>
#include <set>
#include <unordered_map>

// map: árbol RB, O(log n), ordenado por clave
map<string,int> edades;
edades["Ana"] = 25; edades["Luis"] = 30;

// set: conjunto sin duplicados, O(log n)
set<int> s = {5,3,8,1,3};  // almacena {1,3,5,8}

// unordered_map: tabla hash, O(1) promedio
unordered_map<string,int> cuenta;
cuenta["hola"]++; cuenta["hola"]++;
cout << cuenta["hola"]; // 2

Iteradores y range-based for

vector<int> nums = {10,20,30,40};

// Iterador explícito
for (auto it = nums.begin(); it != nums.end(); ++it)
    cout << *it << " ";

// Range-based for — más limpio (C++11)
for (int x : nums)
    cout << x << " ";

Algoritmos de la STL

#include <algorithm>
#include <numeric>

vector<int> v = {5,2,8,1,9,3};
sort(v.begin(), v.end());              // {1,2,3,5,8,9}  O(n log n)
auto it  = find(v.begin(),v.end(),8);  // iterador al 8  O(n)
int  cnt = count(v.begin(),v.end(),5); // 1              O(n)
reverse(v.begin(), v.end());           // {9,8,5,3,2,1}  O(n)
int  s   = accumulate(v.begin(),v.end(),0); // suma=28   O(n)
