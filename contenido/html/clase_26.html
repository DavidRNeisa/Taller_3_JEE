Clase 26: Recorridos en Grafos — DFS y BFS

Explorar un grafo sistemáticamente: en profundidad y en anchura

El problema del recorrido

En un árbol, el recorrido es natural: comenzamos en la raíz. En un grafo, cualquier nodo puede ser el inicio, puede haber ciclos, y el grafo puede no ser conexo. Necesitamos un arreglo de visitados para evitar procesar el mismo nodo dos veces.

DFS — Búsqueda en Profundidad

Explora tan lejos como sea posible por cada rama antes de retroceder. Usa una pila (implícita con recursión).

vector<vector<int>> adj;  // lista de adyacencia
vector<bool> visitado;

void DFS(int v) {
    visitado[v] = true;
    cout << v << " ";
    for (int u : adj[v]) {
        if (!visitado[u])
            DFS(u);
    }
}

// DFS iterativo (sin recursión, con pila explícita)
void DFS_iter(int inicio) {
    stack<int> pila;
    pila.push(inicio);
    while (!pila.empty()) {
        int v = pila.top(); pila.pop();
        if (visitado[v]) continue;
        visitado[v] = true;
        cout << v << " ";
        for (int u : adj[v]) if (!visitado[u]) pila.push(u);
    }
}
// Complejidad: O(V + E)

BFS — Búsqueda en Anchura

Explora por niveles: primero los vecinos directos, luego los vecinos de vecinos, etc. Usa una cola. Encuentra el camino más corto (en número de aristas) en grafos no ponderados.

void BFS(int inicio) {
    queue<int> cola;
    visitado[inicio] = true;
    cola.push(inicio);
    while (!cola.empty()) {
        int v = cola.front(); cola.pop();
        cout << v << " ";
        for (int u : adj[v]) {
            if (!visitado[u]) {
                visitado[u] = true;
                cola.push(u);
            }
        }
    }
}
// Complejidad: O(V + E)

Encontrar el camino más corto con BFS

vector<int> caminoMasCorto(int origen, int destino, int V) {
    vector<int> padre(V, -1);
    vector<bool> vis(V, false);
    queue<int> q;
    vis[origen]=true; q.push(origen);
    while (!q.empty()) {
        int v=q.front(); q.pop();
        if (v==destino) break;
        for (int u : adj[v]) if (!vis[u]) { vis[u]=true; padre[u]=v; q.push(u); }
    }
    // Reconstruir el camino
    vector<int> camino;
    for (int v=destino; v!=-1; v=padre[v]) camino.push_back(v);
    reverse(camino.begin(), camino.end());
    return camino;
}
