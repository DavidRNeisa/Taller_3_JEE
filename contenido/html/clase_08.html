Implementación desde cero: pila, cola y lista enlazada

Pila con arreglo estático

const int CAP = 100;
int datos[CAP];
int cima = -1;

void push(int v) {
    if (cima == CAP-1) { cerr << "Pila llena"; return; }
    datos[++cima] = v;
}
int  pop()       { if (cima==-1) throw runtime_error("Vacía"); return datos[cima--]; }
int  top()       { return datos[cima]; }
bool estaVacia() { return cima == -1; }

Cola circular con arreglo

int frente=0, fin=0, cantidad=0;
int buf[CAP];

void enqueue(int v) {
    if (cantidad==CAP) { cerr << "Cola llena"; return; }
    buf[fin] = v;
    fin = (fin+1) % CAP;   // avance circular
    cantidad++;
}
int dequeue() {
    if (cantidad==0) throw runtime_error("Vacía");
    int v = buf[frente];
    frente = (frente+1) % CAP;
    cantidad--;
    return v;
}

Lista enlazada simple

struct Nodo { int dato; Nodo* sig; Nodo(int d):dato(d),sig(nullptr){} };

struct Lista {
    Nodo* cabeza = nullptr;

    void insertarFrente(int v) {
        Nodo* n = new Nodo(v); n->sig = cabeza; cabeza = n;
    }
    void insertarFinal(int v) {
        Nodo* n = new Nodo(v);
        if (!cabeza) { cabeza=n; return; }
        Nodo* a = cabeza;
        while (a->sig) a = a->sig;
        a->sig = n;
    }
    void eliminar(int v) {
        if (!cabeza) return;
        if (cabeza->dato==v) { Nodo* t=cabeza; cabeza=cabeza->sig; delete t; return; }
        Nodo* a = cabeza;
        while (a->sig && a->sig->dato!=v) a=a->sig;
        if (a->sig) { Nodo* t=a->sig; a->sig=t->sig; delete t; }
    }
    void imprimir() {
        for (Nodo* n=cabeza; n; n=n->sig) cout << n->dato << " → ";
        cout << "NULL\n";
    }
};

Aplicación: verificar paréntesis balanceados

bool balanceado(const string& s) {
    stack<char> p;
    for (char c : s) {
        if (c=='(' || c=='[' || c=='{') p.push(c);
        else if (c==')' || c==']' || c=='}') {
            if (p.empty()) return false;
            char t = p.top(); p.pop();
            if ((c==')'&&t!='(') || (c==']'&&t!='[') || (c=='}'&&t!='{'))
                return false;
        }
    }
    return p.empty();
}
// "({[]})"  → true
// "([)]"    → false
Complejidades: Operaciones de pila/cola con arreglo → O(1). Insertar al final en lista sin puntero a cola → O(n). Con puntero a cola → O(1).
