Clase 28: Prim y Dijkstra

Árbol de expansión mínima y caminos más cortos desde una fuente

Algoritmo de Prim — Árbol de Expansión Mínima

Dado un grafo conexo ponderado no dirigido, el árbol de expansión mínima (MST) es el subconjunto de aristas que conecta todos los vértices con el mínimo peso total. Prim construye el MST creciendo un árbol desde un nodo inicial.

// Prim con min-heap — O((V+E) log V)
void prim(vector<vector<pair<int,int>>>& adj, int V) {
    vector<int> dist(V, INT_MAX), padre(V,-1);
    vector<bool> enMST(V, false);
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;

    dist[0] = 0; pq.push({0, 0});
    while (!pq.empty()) {
        int u = pq.top().second; pq.pop();
        if (enMST[u]) continue;
        enMST[u] = true;
        for (auto [w, v] : adj[u]) {
            if (!enMST[v] && w < dist[v]) {
                dist[v]=w; padre[v]=u;
                pq.push({w, v});
            }
        }
    }
    int pesoTotal = 0;
    for (int v=1; v<V; v++) {
        cout << padre[v] << "-" << v << " (" << dist[v] << ")\n";
        pesoTotal += dist[v];
    }
    cout << "Peso MST: " << pesoTotal << "\n";
}

Algoritmo de Dijkstra — Camino Más Corto

Dado un grafo ponderado con pesos no negativos, encuentra el camino más corto desde un nodo fuente a todos los demás. Utiliza una estrategia voraz con cola de prioridad.

// Dijkstra con min-heap — O((V+E) log V)
vector<int> dijkstra(vector<vector<pair<int,int>>>& adj, int origen, int V) {
    vector<int> dist(V, INT_MAX);
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;

    dist[origen] = 0;
    pq.push({0, origen});

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;  // versión desactualizada, ignorar

        for (auto [w, v] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}
// dist[v] = distancia mínima desde origen hasta v

Reconstruir el camino

vector<int> padre(V, -1);
// Dentro de Dijkstra, cuando actualizamos dist[v]: padre[v] = u;
// Para reconstruir:
vector<int> reconstruir(int destino, vector<int>& padre) {
    vector<int> camino;
    for (int v=destino; v!=-1; v=padre[v]) camino.push_back(v);
    reverse(camino.begin(), camino.end());
    return camino;
}
