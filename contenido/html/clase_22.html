Implementación de la clase BST

#include <iostream>
#include <queue>
using namespace std;

struct Nodo {
    int v; Nodo *izq,*der;
    Nodo(int x):v(x),izq(nullptr),der(nullptr){}
};

class BST {
    Nodo* raiz = nullptr;

    Nodo* insertar(Nodo* r, int val) {
        if (!r) return new Nodo(val);
        if (val < r->v) r->izq = insertar(r->izq, val);
        else if (val > r->v) r->der = insertar(r->der, val);
        return r;
    }
    bool buscar(Nodo* r, int val) {
        if (!r) return false;
        if (val==r->v) return true;
        return val<r->v ? buscar(r->izq,val) : buscar(r->der,val);
    }
    Nodo* minNodo(Nodo* r) { while(r->izq) r=r->izq; return r; }
    Nodo* eliminar(Nodo* r, int val) {
        if (!r) return nullptr;
        if (val<r->v)      r->izq=eliminar(r->izq,val);
        else if (val>r->v) r->der=eliminar(r->der,val);
        else {
            if (!r->izq && !r->der) { delete r; return nullptr; }
            if (!r->izq) { Nodo* t=r->der; delete r; return t; }
            if (!r->der) { Nodo* t=r->izq; delete r; return t; }
            Nodo* s=minNodo(r->der);
            r->v=s->v;
            r->der=eliminar(r->der,s->v);
        }
        return r;
    }
    void inorden(Nodo* r) {
        if(!r) return; inorden(r->izq); cout<<r->v<<" "; inorden(r->der);
    }
    int altura(Nodo* r) {
        if(!r) return 0; return 1+max(altura(r->izq),altura(r->der));
    }
    void porNiveles(Nodo* r) {
        if(!r) return;
        queue<Nodo*> q; q.push(r);
        while(!q.empty()){
            Nodo* n=q.front();q.pop();
            cout<<n->v<<" ";
            if(n->izq) q.push(n->izq);
            if(n->der) q.push(n->der);
        }
    }
public:
    void insertar(int v)    { raiz=insertar(raiz,v); }
    bool buscar(int v)      { return buscar(raiz,v); }
    void eliminar(int v)    { raiz=eliminar(raiz,v); }
    void imprimirInorden()  { inorden(raiz); cout<<"\n"; }
    void imprimirBFS()      { porNiveles(raiz); cout<<"\n"; }
    int  getAltura()        { return altura(raiz); }
};

Prueba de la implementación

BST arbol;
for (int v : {10,5,15,3,7,12,20}) arbol.insertar(v);

arbol.imprimirInorden();  // 3 5 7 10 12 15 20
arbol.imprimirBFS();      // 10 5 15 3 7 12 20
cout << arbol.getAltura(); // 3
cout << arbol.buscar(7);   // 1 (true)
arbol.eliminar(5);
arbol.imprimirInorden();   // 3 7 10 12 15 20
