Ejercicio 2: Altura y balance de un árbol

int altura(Nodo* r) {
    if (!r) return 0;
    return 1 + max(altura(r->izq), altura(r->der));
}
int balance(Nodo* r) {
    if (!r) return 0;
    return altura(r->der) - altura(r->izq);
}
// Para el árbol del ejercicio 1:
// altura(raíz=10) = 3
// balance(raíz=10) = altura(15)-altura(5) = 2-2 = 0  ✓ balanceado

Ejercicio 3: ¿Es el árbol un AVL?

bool esAVL(Nodo* r) {
    if (!r) return true;
    int b = balance(r);
    if (b < -1 || b > 1) return false;
    return esAVL(r->izq) && esAVL(r->der);
}
// El árbol del ej. 1 es AVL (todos los balances entre -1 y 1)
