Clase 18: Taller de Árboles

Práctica de recorridos, altura, conteo y niveles

Los tres recorridos en profundidad (DFS)

// Inorden: Izq → Raíz → Der
// Resultado en BST: valores en orden ascendente
void inorden(Nodo* r) {
    if (!r) return;
    inorden(r->izq); cout << r->v << " "; inorden(r->der);
}

// Preorden: Raíz → Izq → Der
// Útil para serializar/copiar el árbol
void preorden(Nodo* r) {
    if (!r) return;
    cout << r->v << " "; preorden(r->izq); preorden(r->der);
}

// Postorden: Izq → Der → Raíz
// Útil para eliminar el árbol (los hijos antes que el padre)
void postorden(Nodo* r) {
    if (!r) return;
    postorden(r->izq); postorden(r->der); cout << r->v << " ";
}

Recorrido por niveles (BFS)

#include <queue>
void porNiveles(Nodo* raiz) {
    if (!raiz) return;
    queue<Nodo*> q;
    q.push(raiz);
    while (!q.empty()) {
        Nodo* n = q.front(); q.pop();
        cout << n->v << " ";
        if (n->izq) q.push(n->izq);
        if (n->der) q.push(n->der);
    }
}
// Árbol de ejemplo → salida BFS: 10 5 15 3 7 12 20

Calcular la altura

int altura(Nodo* r) {
    if (!r) return 0;
    return 1 + max(altura(r->izq), altura(r->der));
}
// Complejidad: O(n) — debe visitar todos los nodos

Contar nodos

int contar(Nodo* r) {
    if (!r) return 0;
    return 1 + contar(r->izq) + contar(r->der);
}

Verificar si es BST

bool esBST(Nodo* r, int minVal, int maxVal) {
    if (!r) return true;
    if (r->v <= minVal || r->v >= maxVal) return false;
    return esBST(r->izq, minVal, r->v) &&
           esBST(r->der, r->v, maxVal);
}
// Llamada inicial: esBST(raiz, INT_MIN, INT_MAX)
