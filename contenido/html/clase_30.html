Clase 30: Taller de Grafos

Resolver problemas reales con DFS, BFS y Dijkstra

Problema 1: Componentes conexas con DFS

// ¿Cuántas islas (componentes) hay en el grafo?
int contarComponentes(int V, vector<vector<int>>& adj) {
    vector<bool> vis(V, false);
    int componentes = 0;
    for (int i=0; i<V; i++) {
        if (!vis[i]) {
            DFS(i, adj, vis);
            componentes++;
        }
    }
    return componentes;
}
void DFS(int v, vector<vector<int>>& adj, vector<bool>& vis) {
    vis[v]=true;
    for (int u : adj[v]) if (!vis[u]) DFS(u, adj, vis);
}

Problema 2: Camino más corto en laberinto con BFS

// Laberinto: 0=libre, 1=pared. Encontrar mínimo pasos de (0,0) a (n-1,m-1)
int bfsLaberinto(vector<vector<int>>& grid) {
    int n=grid.size(), m=grid[0].size();
    vector<vector<int>> dist(n, vector<int>(m,-1));
    queue<pair<int,int>> q;
    dist[0][0]=0; q.push({0,0});
    int dx[]={0,0,1,-1}, dy[]={1,-1,0,0};
    while (!q.empty()) {
        auto [x,y]=q.front(); q.pop();
        if (x==n-1 && y==m-1) return dist[x][y];
        for (int d=0;d<4;d++) {
            int nx=x+dx[d], ny=y+dy[d];
            if (nx>=0&&nx<n&&ny>=0&&ny<m&&grid[nx][ny]==0&&dist[nx][ny]==-1) {
                dist[nx][ny]=dist[x][y]+1;
                q.push({nx,ny});
            }
        }
    }
    return -1;  // sin solución
}

Problema 3: Detectar ciclo con DFS

bool hayCiclo_DFS(int v, int padre, vector<vector<int>>& adj, vector<bool>& vis) {
    vis[v]=true;
    for (int u : adj[v]) {
        if (!vis[u]) {
            if (hayCiclo_DFS(u, v, adj, vis)) return true;
        } else if (u!=padre) return true;  // arista trasera → ciclo
    }
    return false;
}

Problema 4: Dijkstra desde múltiples fuentes

// Distancia mínima desde cualquier nodo fuente a cada nodo
vector<int> dijkstraMultiFuente(vector<int> fuentes, int V,
                                vector<vector<pair<int,int>>>& adj) {
    vector<int> dist(V, INT_MAX);
    priority_queue<pair<int,int>,vector<pair<int,int>>,greater<>> pq;
    for (int f : fuentes) { dist[f]=0; pq.push({0,f}); }
    while (!pq.empty()) {
        auto [d,u]=pq.top(); pq.pop();
        if (d>dist[u]) continue;
        for (auto [w,v]:adj[u]) if (dist[u]+w<dist[v]) {
            dist[v]=dist[u]+w; pq.push({dist[v],v});
        }
    }
    return dist;
}
Truco: el Dijkstra multi-fuente es equivalente a añadir un nodo virtual conectado a todas las fuentes con costo 0, y correr Dijkstra desde ese nodo virtual.
