Algoritmo de Kruskal — MST por aristas

Kruskal construye el MST ordenando las aristas por peso y añadiéndolas si no crean un ciclo. Usa la estructura Union-Find para detectar ciclos eficientemente.

// Union-Find con compresión de caminos y unión por rango
struct UnionFind {
    vector<int> padre, rango;
    UnionFind(int n): padre(n), rango(n,0) {
        iota(padre.begin(), padre.end(), 0);
    }
    int encontrar(int x) {
        if (padre[x]!=x) padre[x]=encontrar(padre[x]);  // compresión
        return padre[x];
    }
    bool unir(int x, int y) {
        int px=encontrar(x), py=encontrar(y);
        if (px==py) return false;  // mismo componente: ciclo
        if (rango[px] < rango[py]) swap(px,py);
        padre[py]=px;
        if (rango[px]==rango[py]) rango[px]++;
        return true;
    }
};

int kruskal(int V, vector<tuple<int,int,int>>& aristas) {
    sort(aristas.begin(), aristas.end());  // ordenar por peso
    UnionFind uf(V);
    int pesoTotal=0, aristasAgregadas=0;
    for (auto [w,u,v] : aristas) {
        if (uf.unir(u,v)) {
            cout << u << "-" << v << " (" << w << ")\n";
            pesoTotal += w;
            if (++aristasAgregadas == V-1) break;  // MST completo
        }
    }
    return pesoTotal;
}
// Complejidad: O(E log E) dominada por el ordenamiento

Algoritmo de Floyd-Warshall — Todos los pares

Calcula el camino más corto entre todos los pares de nodos, incluyendo con pesos negativos (pero sin ciclos negativos). Complejidad: O(V³).

void floydWarshall(vector<vector<int>>& dist, int V) {
    // Inicializar: dist[i][j] = peso de arista i→j, INF si no existe, 0 en diagonal

    for (int k=0; k<V; k++)          // nodo intermedio
        for (int i=0; i<V; i++)      // origen
            for (int j=0; j<V; j++)  // destino
                if (dist[i][k] != INF && dist[k][j] != INF)
                    dist[i][j] = min(dist[i][j], dist[i][k]+dist[k][j]);

    // Detectar ciclos negativos: si dist[i][i] < 0 → hay ciclo negativo
}
// dist[i][j] = distancia mínima de i a j tras el algoritmo
